Sprite

Simplifies drawing of a sprite, it has the same properties as the parameter object that is used when calling the draw function on SpriteBatch.

var Sprite = require('lib/sprite.js').Sprite;

// var spriteBatch = new SpriteBatch(...)
var sprite = new Sprite(spriteBatch, 'sprite.png');

// Set the position for the sprite.
sprite.position.x = 100;
sprite.position.y = 200;

spriteBatch.begin();
sprite.draw();
spriteBatch.end();
